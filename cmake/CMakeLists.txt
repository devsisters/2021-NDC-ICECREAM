project(ndc)
cmake_minimum_required(VERSION 3.18)

# set icecream's compiler wrapper
execute_process(COMMAND brew --prefix devsisters-icecream OUTPUT_VARIABLE ICE_PATH)
string(REGEX REPLACE "\n$" "" ICE_PATH ${ICE_PATH})
set(ICECC "${ICE_PATH}/libexec/icecc/bin/clang")
set(ICECXX "${ICE_PATH}/libexec/icecc/bin/clang++")

# set environments
set(ICE_ENV_PATH "${CMAKE_SOURCE_DIR}/../icecream-environments")
if(NOT EXISTS "${ICE_ENV_PATH}/native_mac.tar.gz")
  message(FATAL_ERROR "You don't have ${ICE_ENV_PATH}/native_mac.tar.gz")
endif()
if(NOT EXISTS "${ICE_ENV_PATH}/darwin19.tar.gz")
  message(FATAL_ERROR "You don't have ${ICE_ENV_PATH}/darwin19_mac.tar.gz")
endif()
set(ICECC_VERSION "${ICE_ENV_PATH}/native.tar.gz,Darwin19_x86_64:${ICE_ENV_PATH}/darwin19_mac.tar.gz")

# set compiler wrapper
configure_file(${CMAKE_SOURCE_DIR}/cc_wrap.in cc_wrap)
configure_file(${CMAKE_SOURCE_DIR}/cxx_wrap.in cxx_wrap)
execute_process(COMMAND chmod a+rx "${CMAKE_BINARY_DIR}/cc_wrap" "${CMAKE_BINARY_DIR}/cxx_wrap")
set(CMAKE_C_COMPILER_LAUNCHER "${CMAKE_BINARY_DIR}/cc_wrap")
set(CMAKE_CXX_COMPILER_LAUNCHER "${CMAKE_BINARY_DIR}/cxx_wrap")

add_executable(main)

target_compile_options(main PRIVATE
		$<$<COMPILE_LANGUAGE:CXX>:-std=c++14>
    "-include" "${CMAKE_SOURCE_DIR}/../cpp/pch.h"
    "-target" "x86_64-apple-darwin19"
    "-mmacosx-version-min=10.15"
)
